{{/* Documents a reference property or object, writing Markdown for the item. The parameters are expected to include: - `property` - Mandatory. The reference's schema data - `page` - Mandatory. The hugo Page object for the webpage the documentation is being written on, for resolving references. - `_level` - Optional. The current hierarchy level of the page. Used for determining what heading level to write. - `_id` - Optional. The ID this property belongs to. - `_skipMoreInfo` - Optional. Indicates whether the "For more information..." note should be included. */}} {{- $params := . -}} {{- $referenceProperty := merge $params.property (dict "_asReference" true) -}} {{- $skipMoreInfo := default false $params._skipMoreInfo -}} {{- $current_level := add $referenceProperty._level 1 -}} {{/* Build the basic list of render strings for the reference */}} {{- $renderStrings := slice (partial "schematize/utils/headings/property" $params) | append (partial "schematize/utils/summary" $params) -}} {{/* By default, add a link to the referenced schema docs for more information */}} {{- if not $skipMoreInfo -}} {{- $title := $referenceProperty.title -}} {{- $refValue := index $referenceProperty "$id" -}} {{- $refParams := dict "page" $params.page "target" "page" "ref" $refValue -}} {{- $reference := partial "schematize/utils/references/resolve/path" $refParams -}} {{- $moreInfo := printf "For more information, see [%s](%s)" $title $reference -}} {{- $renderStrings = $renderStrings | append $moreInfo -}} {{- end -}} {{/* Render */}} {{- delimit $renderStrings "\n\n" | safeHTML -}}
