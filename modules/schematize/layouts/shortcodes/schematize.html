{{/* This partial writes the documentation for a JSON schema defined using the schematize helper properties as Data for a Hugo site. */}} {{- $params := . -}} {{- $page := $params.Page -}} {{- $schema_path := $params.Get "path" | default ($params.Get 0) | default $page.Params.Schematize -}} {{- $schema_root := $params.Get "root" | default ($params.Get 1) | default $page.Params.SchematizeRoot | default site.Params.Schematize.SchemaRoot -}} {{- with $schema_path -}} {{/* Use the last segment in the schema path as the schema's id */}} {{- $id := index (split $schema_path "." | last 1) 0 -}} {{/* Retrieve the schema data by iterating through the dot path of the combined root and path */}} {{- with $schema_root -}} {{- $schema_path = printf "%s.%s" $schema_root $schema_path -}} {{- end -}} {{- $schemaData := site.Data -}} {{- range $level := split $schema_path "." -}} {{- $schemaData = index $schemaData $level -}} {{- end -}} {{/* Merge the id into the schema data for the docs partials to use */}} {{- $schemaData = merge $schemaData (dict "_id" $id) -}} {{- if eq $schemaData.type "object" -}} {{ partial "schematize/docs/object" (dict "property" $schemaData "page" $page) }} {{- else if eq $schemaData.type "array" -}} {{- $property := merge $schemaData (dict "_level" 0) -}} {{ partial "schematize/docs/array" (dict "property" $property "page" $page) }} {{- else if eq $schemaData.type nil -}} {{- warnf "Couldn't find schema for path '%s' on page '%s'" $schema_path $page.Title -}} {{- else -}} {{- warnf "Unhandled base schema type %s: %#v" $schemaData.type $schemaData -}} {{- highlight ($schemaData | jsonify (dict "indent" " ")) "json" -}} {{- end -}} {{- end -}}
