{{- $Context := . -}} {{- $PlatenConfig := partialCached "platen/param/getKey" "platen" "platen" -}} {{- $renderedHeading := "" -}} {{/* Use renderers from features first */}} {{- range $feature := $PlatenConfig.features -}} {{- if and $feature.enabled (eq "" $renderedHeading) -}} {{- with $feature.partials.renderers.heading -}} {{- $renderedHeading = partial . $Context }} {{- end -}} {{- end -}} {{- end -}} {{/* Use arbitrary renderers if no feature applied */}} {{- range $markup := $PlatenConfig.markup -}} {{- if and $markup.enabled (eq "" $renderedHeading) -}} {{- with $markup.partials.renderers.heading -}} {{- $renderedHeading = partial . $Context }} {{- end -}} {{- end -}} {{- end -}} {{/* Render the heading or use the default renderer if needed */}} {{- with $renderedHeading -}} {{- safeHTML $renderedHeading -}} {{- else -}} {{- partial "platen/markup/_default/heading" $Context -}} {{- end -}}
