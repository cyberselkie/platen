{{/* Partial to generate page name from Title or File name. Accepts Page as context */}} {{- $Params := . -}} {{- $context := . -}} {{- $title := "" -}} {{- $ConfigKey := "platen.features" -}} {{- $Features := partialCached "platen/param/getKey" $ConfigKey $ConfigKey -}} {{- if reflect.IsMap $Params -}} {{- $context = $Params.Context -}} {{- end -}} {{- range $feature, $settings := $Features -}} {{- if and (eq "" $title) $settings.enabled -}} {{- with $settings.partials.title -}} {{ $title = partial . $Params }} {{- end -}} {{- end -}} {{- end -}} {{- if eq "" $title -}} {{- if $context.Title -}} {{- $title = $context.Title -}} {{- else if and $context.IsSection $context.File -}} {{- $title = path.Base $context.File.Dir | humanize | title -}} {{- else if and $context.IsPage $context.File -}} {{- $title = $context.File.BaseFileName | humanize | title -}} {{- end -}} {{- end -}} {{- return $title -}}
