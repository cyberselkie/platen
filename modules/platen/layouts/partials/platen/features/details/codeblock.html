-}} {{- if $open -}} {{- $detailsAttributes = $detailsAttributes | append "open" -}} {{- end -}} {{- $detailsAttributes = partial "platen/utils/getSafeAttributes" $detailsAttributes -}} {{- $lines := slice (printf "

" "
" -}} {{- $renderedCode = delimit $lines "\n" -}} {{- end -}} {{- return $renderedCode -}}
