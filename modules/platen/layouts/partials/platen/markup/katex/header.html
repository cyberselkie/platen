{{- $Context := . -}} {{- $IntegrityPartial := "platen/utils/getIntegrityAttributes" -}} {{- $ConfigKey := "platen.markup.katex" -}} {{- $KatexConfig := partialCached "platen/param/getKey" $ConfigKey $ConfigKey -}} {{- $loadKatex := $Context.Store.Get "_hasKatex" | default $Context.Params.platen.markup.katex.always_load | default $KatexConfig.always_load -}} {{- if $loadKatex -}} {{/* Determine if mhchem should be loaded too */}} {{- $loadChem := $Context.Store.Get "_hasChem" | default (gt (findRE `\\ce\{(.|\s)+?\}` $Context.RawContent) 0) -}} {{/* Get the KaTeX URLs to use */}} {{- $Version := $KatexConfig.version | default "0.16.4" -}} {{- $UrlPrefix := "https://cdn.jsdelivr.net/npm/katex" -}} {{- $StyleUrl := printf "%s@%s/dist/katex.min.css" $UrlPrefix $Version -}} {{- $LibraryUrl := printf "%s@%s/dist/katex.min.js" $UrlPrefix $Version -}} {{- $ChemistryUrl := printf "%s@%s/dist/contrib/mhchem.min.js" $UrlPrefix $Version -}} {{- $RenderUrl := printf "%s@%s/dist/contrib/auto-render.min.js" $UrlPrefix $Version -}} {{/* Initialize the slice of strings to render */}} {{- $renderStrings := slice -}} {{/* Get the KaTeX style definitions */}} {{- $style := resources.GetRemote $StyleUrl | resources.Copy "katex/katex.min.css" | resources.Fingerprint -}} {{- $styleIntegrity := partialCached $IntegrityPartial $style "KatexStyle" -}} {{- $styleAttributes := slice `rel="stylesheet"` | append (printf `href="%s"` $style.RelPermalink) | append $styleIntegrity -}} {{- $styleAttributes = delimit $styleAttributes " " -}} {{- $style := printf "" $styleAttributes -}} {{- $renderStrings = $renderStrings | append $style -}} {{/* Get the KaTeX API library script */}} {{- $library := resources.GetRemote $LibraryUrl | resources.Copy "katex/katex.min.js" | resources.Fingerprint -}} {{- $libraryIntegrity := partialCached $IntegrityPartial $library "KatexLibrary" -}} {{- $libraryAttributes := slice "defer" | append (printf `src="%s"` $library.RelPermalink) | append $libraryIntegrity -}} {{- $libraryAttributes = delimit $libraryAttributes " " -}} {{- $library := printf "" $libraryAttributes -}} {{- $renderStrings = $renderStrings | append $library -}} {{/* Get the mhchem library script, if needed */}} {{- if $loadChem -}} {{- $chemistry := resources.GetRemote $ChemistryUrl | resources.Copy "katex/mhchem.min.js" | resources.Fingerprint -}} {{- $chemistryIntegrity := partialCached $IntegrityPartial $chemistry "KatexChemistry" -}} {{- $chemistryAttributes := slice "defer" | append (printf `src="%s"` $chemistry.RelPermalink) | append $chemistryIntegrity -}} {{- $chemistryAttributes = delimit $chemistryAttributes " " -}} {{- $chemistry := printf "" $chemistryAttributes -}} {{- $renderStrings = $renderStrings | append $chemistry -}} {{- end -}} {{/* Get the KaTeX render script */}} {{- $render := resources.GetRemote $RenderUrl | resources.Copy "katex/auto-render.min.js" | resources.Fingerprint -}} {{- $renderIntegrity := partialCached $IntegrityPartial $render "KatexRenderer" -}} {{- $renderAttributes := slice "defer" | append (printf `src="%s"` $render.RelPermalink) | append $renderIntegrity | append `onload="renderMathInElement(document.body);"` -}} {{- $renderAttributes = delimit $renderAttributes " " -}} {{- $render := printf "" $renderAttributes -}} {{- $renderStrings = $renderStrings | append $render -}} {{/* Render */}} {{- delimit $renderStrings "\n" | safeHTML -}} {{- end }}
