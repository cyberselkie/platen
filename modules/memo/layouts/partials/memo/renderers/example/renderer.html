{{- $params := . -}} {{- $definition := $params.Inner -}} {{- $type := $params.Type -}} {{- $attributes := $params.Attributes -}} {{- $page := $params.Page -}} {{- $number := add 1 $params.Ordinal -}} {{- $renderedCode := "" -}} {{- with $definition -}} {{- $title := default "Example" $attributes.title -}} {{- $title = printf "%v. %s" $number $title -}} {{- $id := default (lower $title | urlize) $attributes.id -}} {{- $level := default 3 $attributes.level -}} {{/* Handle leading explanatory text, if any */}} {{- $parts := split $definition "" -}} {{- $info := "" -}} {{- $code := "" -}} {{- if eq 2 (len $parts) -}} {{- $info = index $parts 0 -}} {{- $code = index $parts 1 -}} {{- else -}} {{- $code = $definition -}} {{- end -}} {{/* Render the example heading */}} {{- $headingMarker := strings.Repeat $level "#" -}} {{- $subHeadingMarker := strings.Repeat (add 1 $level) "#" -}} {{- $heading := printf "%s %s { #%s }" $headingMarker $title $id -}} {{- $renderedCode = $page.RenderString $heading -}} {{/* If info text is supplied, render it. */}} {{- with $info -}} {{- $renderedCode = printf "%s\n\n%s" $renderedCode ($page.RenderString $info) -}} {{- end -}} {{/* Munge the code to remove extra whitespace */}} {{- $trimParams := dict "Text" $code "TrimEOL" true -}} {{- $mungedCode := partial "platen/utils/getLeadTrimmedText" $trimParams -}} {{/* Render the input subsection */}} {{- $inputHeading := printf "%s Markdown Input { #%s-markdown-input }" $subHeadingMarker $id -}} {{- $inputHeading = $page.RenderString $inputHeading -}} {{- $inputContent := printf "``````markdown\n%s\n``````" $mungedCode -}} {{- $inputContent = $page.RenderString $inputContent -}} {{- $renderedCode = printf "%s\n\n%s\n\n%s" $renderedCode $inputHeading $inputContent -}} {{/* Render the HTML output subsection */}} {{- $htmlHeading := printf "%s HTML Output { #%s-html-output }" $subHeadingMarker $id -}} {{- $htmlHeading = $page.RenderString $htmlHeading -}} {{- $htmlContent := printf "``````html\n%s\n``````" ($page.RenderString $mungedCode) -}} {{- $htmlContent = $page.RenderString $htmlContent -}} {{- $renderedCode = printf "%s\n\n%s\n\n%s" $renderedCode $htmlHeading $htmlContent -}} {{/* Render the actual output subsection */}} {{- $renderHeading := printf "%s Rendered Output { #%s-rendered-output }" $subHeadingMarker $id -}} {{- $renderHeading = $page.RenderString $renderHeading -}} {{- $renderContent := $page.RenderString $mungedCode -}} {{- $renderedCode = printf "%s\n\n%s\n\n%s" $renderedCode $renderHeading $renderContent -}} {{- end -}} {{- return $renderedCode -}}
