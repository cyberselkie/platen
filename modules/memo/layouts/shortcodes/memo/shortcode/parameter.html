{{/* Docs.ShortCode: memo/shortcode/syntax */}} {{- $input := . -}} {{- $pageParams := $input.Page.Params -}} {{- $data := $pageParams.Memo -}} {{/* Process input parameters */}} {{- $name := default ($input.Get 0) ($input.Get "name") -}} {{- if hasPrefix $name "Nested." -}} {{- $data = $data.innertext.nested -}} {{- $nameParts := split $name "." -}} {{- $nestedName := slice -}} {{ range $index, $part := $nameParts -}} {{- if ne 0 $index -}} {{- $nestedName = $nestedName | append $part -}} {{- end -}} {{- end -}} {{- $name = delimit $nestedName "." -}} {{- end -}} {{- with (index $data.parameters (lower $name)) -}} {{- $param := . -}} {{- $type := $param.type -}} {{- $stringsToRender := slice -}} {{- with $type -}} {{- $type = apply (split $type ".") "title" "." -}} {{- $type = delimit $type "." -}} {{- else -}} {{- $type = "Any" -}} {{- end -}} {{- $param = merge $param (dict "type" $type) -}} {{- if not $param.nosuffix -}} {{- $docsSuffix := partial "memo/utils/shortcode/getDocsSuffix" $param -}} {{- $stringsToRender = $stringsToRender | append $docsSuffix -}} {{- end -}} {{- $metadataCodeBlock := partial "memo/utils/shortcode/getMetadataCodeBlock" $param -}} {{- $stringsToRender = $stringsToRender | append $metadataCodeBlock -}} {{- $stringToRender := delimit $stringsToRender "\n\n" -}} {{- $input.Page.RenderString $stringToRender -}} {{- else -}} {{- with $data -}} {{- $definition := $data | jsonify (dict "indent" " ") -}} {{- errorf "Couldn't find parameter '%s' in shortcode definition: %s" $name $definition -}} {{- else -}} {{- errorf "Couldn't find parameter '%s'" $name -}} {{- end -}} {{- end -}}
